\subsection{Grammar}

\begin{grammar}
<Var> ::= Any identifier

<Set> ::= $\varnothing$
\alt <Var>

<Bool> ::= <Set> $\in$ <Set>

<Com> ::= `skip'
\alt `if' <Bool> `then' <Com> `else' <Com>
\alt <Var> `:=' <Set>
\alt <Com> `;' <Com>

\end{grammar}
